package cn.net.wangshifu;

import java.text.SimpleDateFormat;

public class TimeConvertUtil {
static SimpleDateFormat format = new SimpleDateFormat(“yyyy-MM-dd HH:mm:ss:SSS”);
public static String getCurrentTime() {
return format.format(System.currentTimeMillis());
}
}

package cn.net.wangshifu;

import org.springframework.beans.factory.InitializingBean;
import javax.annotation.PostConstruct;

public class TestLoadingOrder implements InitializingBean{

private static String test1;
private String test2;

static {
    test1 = "static";
    System.out.println(TimeConvertUtil.getCurrentTime()+" TestLoadingOrder.static has been executed");
}

public static String getTest1() {
    return test1;
}

public static void setTest1(String test1) {
    TestLoadingOrder.test1 = test1;
}

public String getTest2() {
    return test2;
}

public void setTest2(String test2) {
    this.test2 = test2;
    System.out.println(TimeConvertUtil.getCurrentTime()+" TestLoadingOrder.setTest2 has been executed");
}

public void init() {
    System.out.println(TimeConvertUtil.getCurrentTime()+" TestLoadingOrder.init has been executed");
}

TestLoadingOrder() {
    System.out.println(TimeConvertUtil.getCurrentTime()+" TestLoadingOrder.construct has been executed");
}

public void afterPropertiesSet() throws Exception {
    System.out.println(TimeConvertUtil.getCurrentTime()+" TestLoadingOrder.afterProSertiesSet has been executed");
}

@PostConstruct
public void postConstruct(){
    System.out.println(TimeConvertUtil.getCurrentTime()+" TestLoadingOrder.postConstruct has been executed");
}

}

package cn.net.wangshifu;

import org.springframework.beans.BeansException;
import org.springframework.beans.factory.config.BeanPostProcessor;

public class MyBeanPostProcessor implements BeanPostProcessor {

public Object postProcessBeforeInitialization(Object o, String s) throws BeansException {
    System.out.println(TimeConvertUtil.getCurrentTime()+" postProcessBeforeInitialization :" + o.getClass());
    return o;
}

public Object postProcessAfterInitialization(Object o, String s) throws BeansException {
    System.out.println(TimeConvertUtil.getCurrentTime()+" postProcessAfterInitialization :" + o.getClass());
    return o;
}

}

package cn.net.wangshifu;

import org.springframework.beans.BeansException;
import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;

public class MyBeanFactoryPostProcessor implements BeanFactoryPostProcessor {

public void postProcessBeanFactory(ConfigurableListableBeanFactory configurableListableBeanFactory) throws BeansException {
    System.out.println(TimeConvertUtil.getCurrentTime()+" postProcessBeanFactory has been executed");
}

}

package cn.net.wangshifu;

import org.springframework.context.ApplicationContext;
import org.springframework.context.support.ClassPathXmlApplicationContext;

public class Main {

public static void main(String[] args) {
    ApplicationContext applicationContext = new ClassPathXmlApplicationContext("applicationContext.xml");
    TestLoadingOrder testLoadingOrder = (TestLoadingOrder) applicationContext.getBean("testLoadingOrder");
    System.out.println("test1: " + TestLoadingOrder.getTest1());
    System.out.println("test2: " + testLoadingOrder.getTest2());
}

}
